% Customisation (fonts, sizes, colours)

% Font families and sizes
\def\monofont{\dofont Liberation Mono}
\def\seriffont{\dofont CMU Serif}

\fontsize=6pt
\serifsize=7pt
\font\headlinefont=\seriffont, 8pt!
\font\headerfont=\monofont, 8.5pt!
\def\monofam{%
	\font\rm=\monofont, \fontsize!%
	\font\bf=\monofont/B, \fontsize!%
	\font\it=\monofont/I, \fontsize!%
}
\def\seriffam{%
	\font\rm=\seriffont:mapping=tex-text, \serifsize!%
	\font\bf=\seriffont/B:mapping=tex-text, \serifsize!%
	\font\it=\seriffont/I:mapping=tex-text, \serifsize!%
	\baselineskip=8.5pt%
}

% Colours
\def\typ{\col{0 0 .5}\bf}   % type definition
\def\com{\col{.3 .3 .3}\it} % comment
\def\str{\col{0 .7 0}\rm}   % string literal
\def\dir{\col{.5 0 0}\it}   % preprocessor directive
\def\key{\col{0 0 0}\bf}    % keyword
\def\con{\col{.8 .1 .2}\rm} % constant
\def\tem{\col{.0 .2 .8}\bf} % templatable type
\def\hcol{\col{.0 .0 .0}\rm}   % hash at the start of (almost) each line
