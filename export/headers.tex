% Generic header macro

\def\hgeneric#1#2#3#4{{%
	\vskip#1
	#2
	#3\writetocentry{notebook}{#3|}
	%\vskip#4
}}

\def\chapter#1{\hgeneric{\chapterpreskip}{\chapterfont}{#1}{\chapterpostskip}}
\def\section#1{\hgeneric{\sectionpreskip}{\sectionfont}{#1}{\sectionpostskip}}
\def\subsection#1{\hgeneric{\subsectionpreskip}{\subsectionfont}{#1}{\subsectionpostskip}}
\def\subsubsection#1{\hgeneric{\subsubsectionpreskip}{\subsubsectionfont}{#1}{\subsubsectionpostskip}}
